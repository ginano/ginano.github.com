function foo() {
 
			  var x = 10;
			 
			  (function bar() {
			 
			    log(x); 
			    // 20, 不上10,不是从foo的活动对象上得到的
			 
			    // "x"从链上查找:
			    // AO(bar) - no -> __specialObject(bar) -> no
			    // __specialObject(bar).[[Prototype]] - yes: 20
			 
			  })();
			}
			 
			Object.prototype.x = 20;
			 
			foo();
