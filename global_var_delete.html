// 定义三个全局变量
			//校验未声明的全局变量将成为挂载再window全局对象下的属性，因此可以delete
			var global_var = 1;
			global_novar = 2; // 反面教材
			(function () {
			   global_fromfunc = 3; // 反面教材
			}());
			
			// 试图删除
			log(delete global_var);  
			log(delete global_novar);  
			log(delete global_fromfunc);  
			
			// 测试该删除
			log(typeof global_var);  
			log(typeof global_novar);  
			log(typeof global_fromfunc);  
			
			//但是eval在作用域的问题，在其中声明的变量将直接挂载在eval对应的作用域下的一个属性形式
			//因此此时的变量是可以被删除的，代码如下：
			
			eval('var global_var_eval=4;');
			log(delete global_var_eval);
			
			log(typeof global_var_eval);
